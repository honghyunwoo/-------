from app.utils.exceptions import FontLoadError
from loguru import logger

def load_font_safely(font_path: str, size: int) -> ImageFont:
    """안전한 폰트 로딩"""
    try:
        font = ImageFont.truetype(font_path, size)
        logger.debug(f"Font loaded successfully: {font_path}")
        return font
    except OSError as e:
        logger.warning(f"Font file not found: {font_path} - {e}")
        raise FontLoadError(f"Font file not found: {font_path}") from e
    except Exception as e:
        logger.error(f"Unexpected error loading font: {font_path}", exc_info=True)
        raise FontLoadError(f"Failed to load font: {font_path}") from e
